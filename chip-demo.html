JavaScript API
// Initialize
const chipElement = document.querySelector('.chip')
const chip = new bootstrap.Chip(chipElement)

// Close
chip.close()

// Dispose
chip.dispose()

// Events
chipElement.addEventListener('close.bs.chip', function (event) {
  // Do something before chip closes
})

chipElement.addEventListener('closed.bs.chip', function (event) {
  // Do something after chip is closed
})
